##### Базовые операции выборки

```sql
-- Вернуть все строки из таблицы Article в произвольном порядке
SELECT * FROM Article;

-- Вернуть строки с сортировкой
SELECT * FROM Article ORDER BY id;
SELECT * FROM Article ORDER BY creationTime DESC, id;

-- Вернуть только определенные столбцы
SELECT id, userId FROM Article;

-- Вернуть уникальные значения
SELECT DISTINCT(userId) FROM Article;

-- Вернуть пользователей, которые написали статьи и зарегистрированы не позже недели назад
SELECT * FROM User 
WHERE
    id IN (SELECT userId FROM Article) 
    AND creationTime > NOW() - INTERVAL 7 DAY;

-- Максимальное время регистрации
SELECT MAX(creationTime) FROM User;

-- Последние 10 зарегистрированных пользователей с openId
SELECT * FROM User 
WHERE openId IS NOT NULL 
ORDER BY creationTime DESC 
LIMIT 10;
```

##### Операции изменения данных

```sql
-- Сменить автора статьи
UPDATE Article SET userId = 10 WHERE id = 7;

-- Удалить статьи не первого пользователя за последнюю неделю
DELETE FROM Article 
WHERE userId != 1
    AND creationTime > NOW() - INTERVAL 7 DAY;

-- Вставить новую статью
INSERT INTO `Article` (`userId`, `text`, `creationTime`) 
VALUES (4, 'VK Cup 2019', NOW());
```
